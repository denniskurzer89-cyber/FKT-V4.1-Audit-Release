    <script>
        const apiKey = ""; // Wird zur Laufzeit bereitgestellt
        const waveCanvas = document.getElementById('waveCanvas');
        const waveCtx = waveCanvas.getContext('2d');
        const phaseCanvas = document.getElementById('phaseSpaceCanvas');
        const phaseCtx = phaseCanvas.getContext('2d');
        const radarCanvas = document.getElementById('radarCanvas');
        const radarCtx = radarCanvas.getContext('2d');
        const logContainer = document.getElementById('systemLog');
        const aiContent = document.getElementById('aiContent');
        const aiLoader = document.getElementById('aiLoadingIndicator');
        
        let time = 0;
        let samples = 0; // Start bei 0 für authentische Kohärenz-Analyse
        let phaseHistory = [];

        function resize() {
            [waveCanvas, phaseCanvas, radarCanvas].forEach(c => {
                c.width = c.offsetWidth;
                c.height = c.offsetHeight;
            });
        }
        window.addEventListener('resize', resize);

        function addLog(msg) {
            const p = document.createElement('p');
            p.textContent = `> [${new Date().toLocaleTimeString()}] ${msg}`;
            logContainer.appendChild(p);
            logContainer.scrollTop = logContainer.scrollHeight;
            if (logContainer.children.length > 20) logContainer.removeChild(logContainer.firstChild);
        }

        async function callGemini(prompt, systemPrompt) {
            aiLoader.classList.remove('hidden');
            let retries = 5;
            let delay = 1000;

            while (retries > 0) {
                try {
                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: prompt }] }],
                            systemInstruction: { parts: [{ text: systemPrompt }] },
                            tools: [{ "google_search": {} }]
                        })
                    });
                    const result = await response.json();
                    aiLoader.classList.add('hidden');
                    return result.candidates?.[0]?.content?.parts?.[0]?.text || "Fehler in der Antwort-Extraktion.";
                } catch (error) {
                    retries--;
                    if (retries === 0) {
                        aiLoader.classList.add('hidden');
                        return "KI-Verbindungsfehler nach 5 Versuchen. Prüfen Sie die Master-Synchronisation.";
                    }
                    await new Promise(res => setTimeout(res, delay));
                    delay *= 2;
                }
            }
        }

        async function askGeminiAnalysis() {
            const statusData = `H0: 71.5, Omega_m: 0.33, Gelman-Rubin: 0.0097, Samples: ${samples}, DOI: 10.5281/zenodo.19761019`;
            const systemPrompt = "Du bist eine KI-Schnittstelle für die Fraktale Kausale Theorie (FKT). Interpretiere die physikalischen Daten im Kontext des Kurzer-Prinzips. Antworte präzise, wissenschaftlich und leicht futuristisch auf Deutsch. Nutze Google Search für aktuelle kosmologische Vergleiche.";
            const userPrompt = `Analysiere den aktuellen Systemstatus: ${statusData}. Ist die Kohärenz stabil?`;
            
            aiContent.innerHTML = "✨ Übertrage Daten an KI-Kern...";
            const analysis = await callGemini(userPrompt, systemPrompt);
            aiContent.innerHTML = analysis.replace(/\n/g, '<br>');
            addLog("✨ KI-ANALYSE empfangen.");
        }

        async function predictStability() {
            const systemPrompt = "Du bist ein Prognose-Modul für MCMC-Ketten. Analysiere, ob eine Gelman-Rubin-Metrik von 0,0097 bei fast 2 Mio Samples einen permanenten Lock garantiert.";
            const userPrompt = "Erstelle eine Kurzprognose für die nächsten 10 Mio Samples.";
            
            aiContent.innerHTML = "✨ Berechne Inferenz-Pfad...";
            const prediction = await callGemini(userPrompt, systemPrompt);
            aiContent.innerHTML = "<b>PROGNOSE:</b><br>" + prediction.replace(/\n/g, '<br>');
            addLog("✨ STABILITÄTS-PROGNOSE abgeschlossen.");
        }

        function drawWave() {
            waveCtx.clearRect(0, 0, waveCanvas.width, waveCanvas.height);
            waveCtx.strokeStyle = '#00ff41';
            waveCtx.lineWidth = 2;
            waveCtx.beginPath();
            
            const coherenceFactor = Math.min(1, samples / 100000);
            for(let x = 0; x < waveCanvas.width; x++) {
                const y = (waveCanvas.height / 2) + Math.sin(x * 0.05 + time) * (15 * coherenceFactor);
                if(x === 0) waveCtx.moveTo(x, y);
                else waveCtx.lineTo(x, y);
            }
            waveCtx.stroke();
        }

        function drawPhaseSpace() {
            phaseCtx.fillStyle = 'rgba(0, 0, 0, 0.05)';
            phaseCtx.fillRect(0, 0, phaseCanvas.width, phaseCanvas.height);
            
            const cx = phaseCanvas.width / 2;
            const cy = phaseCanvas.height / 2;
            
            const spread = Math.max(5, 50 * (1 - Math.min(1, samples / 500000)));
            const px = cx + (Math.random() - 0.5) * spread;
            const py = cy + (Math.random() - 0.5) * spread;
            
            phaseHistory.push({x: px, y: py});
            if(phaseHistory.length > 200) phaseHistory.shift();
            
            phaseCtx.fillStyle = '#00ff41';
            phaseHistory.forEach(p => {
                phaseCtx.beginPath();
                phaseCtx.arc(p.x, p.y, 1.5, 0, Math.PI * 2);
                phaseCtx.fill();
            });
            
            phaseCtx.strokeStyle = 'rgba(0, 255, 65, 0.3)';
            phaseCtx.strokeRect(cx - 10, cy - 10, 20, 20);
        }

        function drawRadar() {
            const cx = radarCanvas.width / 2;
            const cy = radarCanvas.height / 2;
            const radius = Math.max(1, Math.min(cx, cy) - 20);
            
            radarCtx.clearRect(0, 0, radarCanvas.width, radarCanvas.height);
            
            radarCtx.strokeStyle = '#113311';
            radarCtx.lineWidth = 1;
            for(let j=1; j<=3; j++) {
                radarCtx.beginPath();
                radarCtx.arc(cx, cy, (radius / 3) * j, 0, Math.PI*2);
                radarCtx.stroke();
            }
            
            const load = Math.min(1, samples / 1998375);
            radarCtx.fillStyle = `rgba(0, 255, 65, ${0.1 + load * 0.2})`;
            radarCtx.strokeStyle = '#00ff41';
            radarCtx.lineWidth = 2;
            radarCtx.beginPath();
            const segments = 5;
            for(let i=0; i<segments; i++) {
                let angle = (i / segments) * Math.PI * 2 - Math.PI/2;
                let r = radius * (0.6 + load * 0.3 + Math.sin(time * 2 + i) * 0.05);
                let x = cx + Math.cos(angle) * r;
                let y = cy + Math.sin(angle) * r;
                if(i===0) radarCtx.moveTo(x, y);
                else radarCtx.lineTo(x, y);
            }
            radarCtx.closePath();
            radarCtx.fill();
            radarCtx.stroke();
            
            const loadBar = document.getElementById('loadBar');
            if(loadBar) loadBar.style.width = (load * 100) + "%";
        }

        function exportAuditData() {
            const auditPayload = {
                model_identifier: "Fraktale Kausale Theorie (FKT)",
                software_version: "v4.4.4-TG-2025",
                export_timestamp: new Date().toISOString(),
                samples: samples,
                metrics: { h0: 71.5, gelman_rubin: 0.0097 }
            };
            const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(auditPayload, null, 2));
            const dlAnchor = document.createElement('a');
            dlAnchor.setAttribute("href", dataStr);
            dlAnchor.setAttribute("download", `FKT_V444_AUDIT.json`);
            dlAnchor.click();
            addLog("EXPORT: Datensatz extrahiert.");
        }

        function loop() {
            time += 0.04;
            drawWave();
            drawPhaseSpace();
            drawRadar();
            
            samples += 850; // Schnellerer Aufbau für Demo-Zwecke
            const sampleCounter = document.getElementById('sampleCount');
            if(sampleCounter) sampleCounter.textContent = Math.floor(samples).toLocaleString('de-DE');
            
            if(Math.random() > 0.99) {
                const logs = ["Kohärenz steigt...", "Deep Lock Prüfung...", "Inferenz-Kette stabil"];
                addLog(logs[Math.floor(Math.random() * logs.length)]);
            }

            requestAnimationFrame(loop);
        }

        window.onload = () => {
            resize();
            loop();
        };
